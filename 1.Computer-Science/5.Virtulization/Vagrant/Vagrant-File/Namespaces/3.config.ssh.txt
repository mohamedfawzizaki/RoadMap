                                                                   config.ssh namespace
                                                                   ========================

 _______________________________________________________________________________________________________________________________________________________________________________________________________________________
|               Property                   |                                             Description                                          |                               Example                                   |
|__________________________________________|__________________________________________________________________________________________________|_________________________________________________________________________|
| config.ssh.host                          | The host (hostname or IP) to SSH into. Defaults to empty (Vagrant figures it out).               |   config.ssh.host = "192.168.33.10"                                     |
| config.ssh.username                      | The username for SSH authentication. Defaults to "vagrant".                                      |   config.ssh.username = "user"                                          |
| config.ssh.remote_user                   | The remote user value for ProxyCommand. Defaults to the value of config.ssh.username.            |   config.ssh.remote_user = "remote_user"                                |
| config.ssh.password                      | The password for SSH authentication (if not using key-based authentication).                     |   config.ssh.password = "secretpassword"                                |
| config.ssh.shell                         | The shell to use for SSH commands. Defaults to /bin/bash -l.                                     |   config.ssh.shell = "/bin/zsh"                                         |
| config.ssh.sudo_command                  | The command used for sudo in SSH. Defaults to sudo -E -H %c.                                     |   config.ssh.sudo_command = "sudo -S %c"                                |
| config.ssh.port                          | The port to SSH into. Defaults to port 22.                                                       |   config.ssh.port = 2222                                                |
| config.ssh.guest_port                    | The port on the guest machine for SSH. Defaults to 22.                                           |   config.ssh.guest_port = 2222                                          |
| config.ssh.proxy_command                 | A command-line command to proxy the SSH connection.                                              |   config.ssh.proxy_command = "ssh -q -W %h:%p user@proxyserver"         |
| config.ssh.compression                   | If false, disables SSH compression. Defaults to true (Compression=yes).                          |   config.ssh.compression = false                                        |
| config.ssh.connect_timeout               | Number of seconds to wait for establishing an SSH connection to the guest. Defaults to 15.       |   config.ssh.connect_timeout = 30                                       |
| config.ssh.config                        | Path to a custom ssh_config file for SSH connection configuration.                               |   config.ssh.config = "/path/to/ssh_config"                             |
| config.ssh.disable_deprecated_algorithms | If true, disables deprecated algorithms like ssh-rsa. Defaults to false.                         |   config.ssh.disable_deprecated_algorithms = true                       |
| config.ssh.dsa_authentication            | If false, disables DSA authentication. Defaults to true.                                         |   config.ssh.dsa_authentication = false                                 |
| config.ssh.export_command_template       | Template used to export environment variables in a Bourne incompatible shell.                    |   config.ssh.export_command_template = 'export %ENV_KEY%="%ENV_VALUE%"' |
| config.ssh.extra_args                    | Extra arguments passed directly into the ssh executable (e.g., reverse tunneling).               |   config.ssh.extra_args = ["-L", "8008:localhost:80"]                   |
|------------------------------------------|--------------------------------------------------------------------------------------------------|-------------------------------------------------------------------------|
| config.ssh.forward_agent                 | Enables agent forwarding for SSH connections. Defaults to false.                                 |   config.ssh.forward_agent = true                                       |
| config.ssh.forward_env                   | Array of host environment variables to forward to the guest.                                     |   config.ssh.forward_env = ["CUSTOM_VAR"]                               |
| config.ssh.forward_x11                   | Enables X11 forwarding over SSH connections. Defaults to false.                                  |   config.ssh.forward_x11 = true                                         |
| config.ssh.insert_key                    | If true, Vagrant will automatically insert an SSH key for authentication. Defaults to true.      |   config.ssh.insert_key = false                                         |
| config.ssh.keep_alive                    | If true, SSH will send keep-alive packets every 5 seconds to maintain the connection.            |   config.ssh.keep_alive = true                                          |
| config.ssh.keys_only                     | If true, only use Vagrant-provided SSH keys, not any keys stored in ssh-agent. Defaults to true. |   config.ssh.keys_only = true                                           |
| config.ssh.key_type                      | The SSH key type when generating a new key to replace the default insecure key.                  |   config.ssh.key_type = :ed25519                                        |
| config.ssh.paranoid                      | Enforces strict host key verification. Deprecated in favor of config.ssh.verify_host_key.        |   config.ssh.paranoid = true                                            |
| config.ssh.private_key_path              | Path(s) to the private key(s) used for SSH authentication.                                       |   config.ssh.private_key_path = "~/.ssh/id_rsa"                         |
| config.ssh.pty                           | If true, use PTY for provisioning. Defaults to false.                                            |   config.ssh.pty = true                                                 |
| config.ssh.verify_host_key               | The host key verification method. Defaults to :never. Options: :accept_new_or_local_tunnel, :accept_new, or :always. | config.ssh.verify_host_key = :accept_new            |
